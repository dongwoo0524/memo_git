void update(String keyword,String SP,String RP) {
		String result = "";
		int searchcnt=0;
		for(int i=0;i<cnt;i++) {
			searchcnt++;
			if(keyword.equals(arrPlane[i][1])) {
				break;
			}
		}
		if(RP == null) {
			arrPlane[searchcnt][3] = SP;
		}else if(SP == null){
			arrPlane[searchcnt][4] = RP;			
		}else if(SP == RP) {
			result = "수정할수 없습니다.";
		}
		else {
			arrPlane[searchcnt][3] = SP;
			arrPlane[searchcnt][4] = RP;
		}
	}
